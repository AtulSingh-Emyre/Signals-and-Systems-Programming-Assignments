for i in fs:
    y=np.sin((2*np.pi*fa*x)/i)
    yC = []
    for Y in y:
        if Y > -1/math.pow(10,14) and Y < 1/math.pow(10,14):
            yC.append(0)
        else:
            yC.append(Y)
    y = yC        
    plt.stem(x,y)

    plt.ylabel('y[n]')
